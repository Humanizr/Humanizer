## Snippet source

```csharp
using System;
using System.Globalization;
using System.Linq;
using System.Runtime.InteropServices;

Console.WriteLine("# Raw MonthNames enumeration for ta and zu-ZA");
Console.WriteLine($"# Platform: {RuntimeInformation.OSDescription}");
Console.WriteLine($"# Framework: {RuntimeInformation.FrameworkDescription}");
Console.WriteLine($"# RID: {RuntimeInformation.RuntimeIdentifier}");
Console.WriteLine($"# Timestamp: {DateTime.UtcNow:O}");
Console.WriteLine();

foreach (var c in new[]{"ta","zu-ZA"})
{
    var culture = CultureInfo.GetCultureInfo(c);
    var dtf = culture.DateTimeFormat;
    var months = dtf.MonthNames.Take(12).ToArray();
    var genitive = dtf.MonthGenitiveNames.Take(12).ToArray();

    Console.WriteLine($"## {c}");
    Console.WriteLine();
    Console.WriteLine("MonthNames (12 entries):");
    for (int i = 0; i < 12; i++)
        Console.WriteLine($"  [{i}] \"{months[i]}\"");
    Console.WriteLine();
    Console.WriteLine("MonthGenitiveNames (12 entries):");
    for (int i = 0; i < 12; i++)
        Console.WriteLine($"  [{i}] \"{genitive[i]}\"");
    Console.WriteLine();
    Console.WriteLine($"MonthNames == MonthGenitiveNames: {months.SequenceEqual(genitive)}");
    Console.WriteLine();
}
```
